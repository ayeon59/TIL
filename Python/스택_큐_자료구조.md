class Stack:
    def __init__(self):
        self.data = []

    def push(self, x):
        self.data.append(x)

    def pop(self):
        if self.is_empty():
            raise IndexError("Stack is empty!")
        return self.data.pop()

    def peek(self):
        if self.is_empty():
            raise IndexError("Stack is empty!")
        return self.data[-1]

    def is_empty(self):
        return len(self.data) == 0

    def size(self):
        return len(self.data)
